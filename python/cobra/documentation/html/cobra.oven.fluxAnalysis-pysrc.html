16      ----------
 
17      model : cobra model
 
18      already_irreversible : bool, optional

28      hot_start = model.optimize(**optimize_kwargs) 
29      # if the problem is infeasible
 
30      if model.solution.f is None: 
31          return 
32      old_f = model.solution.f 
33      old_objective_coefficients = {} 
34      old_lower_bounds = {} 
35      old_upper_bounds = {} 
36      for reaction in model.reactions: 
37          # if the reaction has a nonzero objective coefficient, then
 
38          # the same flux should be maintained through that reaction
 
39          if reaction.objective_coefficient != 0: 
40              old_objective_coefficients[reaction] = \

52      model.optimize(**optimize_kwargs) 
53      # make the model back the way it was

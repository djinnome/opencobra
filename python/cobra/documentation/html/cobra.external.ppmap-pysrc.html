 56      """Split the work of 'function' across the given number of 
 57      processes.  Set 'processes' to None to let Parallel Python 
 58      autodetect the number of children to use. 
 59   
 60      Although the calling semantics should be identical to 
 61      __builtin__.map (even using __builtin__.map to process 
 62      arguments), it differs in that it returns a generator instead of a 
 63      list.  This enables lazy evaluation of the results so that other 
 64      work can be done while the subprocesses are still running. 
 65   
 66      >>> def rangetotal(n): return n, sum(range(n)) 
 67      >>> list(map(rangetotal, range(1, 6))) 
 68      [(1, 0), (2, 1), (3, 3), (4, 6), (5, 10)] 
 69      >>> list(ppmap(1, rangetotal, range(1, 6))) 
 70      [(1, 0), (2, 1), (3, 3), (4, 6), (5, 10)] 
 71      """ 
 72   
 73      # Create a new server if one isn't already initialized 
 74      if not __STATE['server']: 
 75          __STATE['server'] = pp.Server() 
 76       
 77      def submit(*args): 
 78          """Send a job to the server""" 
 79          return __STATE['server'].submit(function, args, globals=globals()) 

 80   
 81      # Merge all the passed-in argument lists together.  This is done 
 82      # that way because as with the map() function, at least one list 
 83      # is required but the rest are optional. 
 84      a = [sequence] 
 85      a.extend(sequences) 
 86      available_processes = sum( __STATE['server'].get_active_nodes().values()) 
 87      # Set the requested level of multi-processing 
 88      if available_processes < processes: 
 89          __STATE['server'].set_ncpus(processes-available_processes or 'autodetect') 
 90   
 91      # First, submit all the jobs.  Then harvest the results as they 
 92      # come available. 
 93      return (subproc() for subproc in __builtin__.map(submit, *a))

 68    build_dendrogram <- function(the_data, distance_method = "pearson", 
 69                                  linkage_method = "complete") { 
 70        #Builds a dendrogram based on the distances between the rows in a matrix 
 71        if (distance_method == "pearson") { 
 72             the_dist = pearsonDistance(t(the_data)) 
 73        } else { 
 74             the_dist = dist(the_data, method = distance_method) 
 75        } 
 76        return(as.dendrogram(hclust(the_dist, method = linkage_method)))  
 77   
 78    }
